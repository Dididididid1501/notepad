<script>
  (function() {
    // ---- 1. Определение мобильного устройства (user-agent + touch hints) ----
    function isMobileDevice() {
      // Проверка по user-agent на основные мобильные платформы
      const userAgent = navigator.userAgent || navigator.vendor || window.opera;
      const mobileRegex = /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini|Mobile|Windows Phone/i;
      if (mobileRegex.test(userAgent)) return true;
      
      // Доп. проверка: ширина экрана и поддержка touch (часто у планшетов/телефонов)
      if ('ontouchstart' in window || navigator.maxTouchPoints > 0) {
        // но это может быть и у современных лэптопов с сенсорным экраном.
        // все же оставим дополнительную эвристику: если ширина <= 1024px и есть touch — считаем мобильным/планшетом
        if (window.innerWidth <= 1024) return true;
      }
      return false;
    }

    const isMobile = isMobileDevice();
    const desktopDiv = document.getElementById('desktopBlock');
    const mobileDiv = document.getElementById('mobileBlock');
    const appRoot = document.getElementById('appRoot');

    // Показать заглушку для десктопа / показать интерфейс для мобильных
    if (!isMobile) {
      // Десктопный режим: показываем заглушку, скрываем мобильный блок
      desktopDiv.classList.remove('hidden');
      mobileDiv.classList.add('hidden');
    } else {
      // Мобильное устройство: скрываем заглушку, показываем основной функционал
      desktopDiv.classList.add('hidden');
      mobileDiv.classList.remove('hidden');
      
      // ---- 2. Инициализация Speech Recognition на мобильном устройстве ----
      // Проверка поддержки API распознавания речи
      const SpeechRecognitionAPI = window.SpeechRecognition || window.webkitSpeechRecognition;
      let recognition = null;
      let isRecording = false;
      
      // DOM элементы мобильного блока
      const recordBtn = document.getElementById('recordButton');
      const statusMsgDiv = document.getElementById('statusMessage');
      const transcriptDisplay = document.getElementById('transcriptDisplay');
      const clearBtn = document.getElementById('clearButton');
      
      // Текущий накопленный текст (последняя распознанная фраза, но по ТЗ просто выводим текст,
      // будем заменять текущий текст новым. Так же добавим опцию накопления? лучше замена, как удобно.
      // Но пользователь может говорить несколько раз — будет перезаписывать предыдущий результат.
      // Сделаем замену при каждом новом успешном распознавании. Иначе если несколько предложений подряд,
      // пользователь может нажать "очистить" или записывать снова. Но можно также добавить режим добавления,
      // но условие: расшифровка в текст --- вывод текста на экран. Будем просто заменять последний результат.
      let currentTranscript = '';
      
      function updateTranscriptUI(text) {
        if (!text || text.trim() === '') {
          transcriptDisplay.innerHTML = '<span class="empty-placeholder">(Здесь появится текст после голосовой команды)</span>';
        } else {
          transcriptDisplay.innerText = text;
        }
      }
      
      function setStatus(message, isError = false) {
        statusMsgDiv.innerHTML = message;
        if (isError) {
          statusMsgDiv.style.backgroundColor = "#ffe6e5";
          statusMsgDiv.style.color = "#b91c1c";
        } else {
          statusMsgDiv.style.backgroundColor = "#f0f4fa";
          statusMsgDiv.style.color = "#2c3e66";
        }
      }
      
      // Очистка текста
      clearBtn.addEventListener('click', () => {
        currentTranscript = '';
        updateTranscriptUI('');
        setStatus('🧹 Текст очищен. Готов к записи');
      });
      
      // Функция остановки записи (если активна)
      function stopRecording() {
        if (recognition && isRecording) {
          try {
            recognition.stop();
          } catch(e) {
            // если уже остановлено, игнорируем
          }
        }
        // Сброс визуального состояния кнопки
        if (recordBtn) {
          recordBtn.classList.remove('active');
          const micSpan = recordBtn.querySelector('.mic-icon');
          if (micSpan) micSpan.innerHTML = '🎤';
        }
        isRecording = false;
        // Не меняем статус на "готов", если только не вызвано отдельно (оставим на обработчике end)
      }
      
      // Инициализация распознавателя с параметрами
      function initRecognition() {
        if (!SpeechRecognitionAPI) {
          setStatus('❌ Ваш браузер не поддерживает распознавание речи. Попробуйте Chrome / Safari на Android или iOS.', true);
          if(recordBtn) recordBtn.disabled = true;
          return null;
        }
        const newRecog = new SpeechRecognitionAPI();
        newRecog.continuous = false;       // однократное распознавание после паузы (удобно)
        newRecog.interimResults = false;   // только финальные результаты (чистый текст)
        newRecog.lang = 'ru-RU';           // русский язык, хорошо для распознавания
        newRecog.maxAlternatives = 1;
        return newRecog;
      }
      
      // Обработчики событий для конкретного экземпляра
      function bindRecognitionEvents(recogInstance) {
        // Когда распознавание успешно
        recogInstance.onresult = (event) => {
          const lastResult = event.results[event.results.length - 1];
          if (lastResult && lastResult.isFinal) {
            const recognizedText = lastResult[0].transcript;
            if (recognizedText && recognizedText.trim().length > 0) {
              currentTranscript = recognizedText;
              updateTranscriptUI(currentTranscript);
              setStatus(`✅ Распознано: "${recognizedText.substring(0, 50)}${recognizedText.length > 50 ? '…' : ''}"`);
            } else {
              setStatus('⚠️ Не удалось распознать речь. Попробуйте еще раз.', false);
            }
          }
        };
        
        recogInstance.onerror = (event) => {
          console.warn('SpeechRecognition error:', event.error);
          let errorMessage = '';
          switch(event.error) {
            case 'not-allowed':
              errorMessage = '🎙️ Доступ к микрофону запрещён. Разрешите доступ в настройках браузера.';
              break;
            case 'no-speech':
              errorMessage = '🤐 Речь не обнаружена. Попробуйте говорить громче.';
              break;
            case 'audio-capture':
              errorMessage = '🎧 Не найден микрофон. Подключите гарнитуру или встроенный микрофон.';
              break;
            case 'network':
              errorMessage = '🌐 Ошибка сети. Проверьте соединение.';
              break;
            default:
              errorMessage = `Ошибка: ${event.error}`;
          }
          setStatus(`❌ ${errorMessage}`, true);
          // останавливаем запись и сбрасываем кнопку
          if (isRecording) {
            if (recordBtn) recordBtn.classList.remove('active');
            const micSpan = recordBtn?.querySelector('.mic-icon');
            if (micSpan) micSpan.innerHTML = '🎤';
            isRecording = false;
          }
        };
        
        recogInstance.onend = () => {
          // Распознавание завершилось (автоматически после паузы или при вызове stop)
          if (isRecording) {
            // если isRecording true, значит, это не было ручной остановкой из кнопки?
            // В любом случае завершаем сессию
            if (recordBtn) recordBtn.classList.remove('active');
            const micSpan = recordBtn?.querySelector('.mic-icon');
            if (micSpan) micSpan.innerHTML = '🎤';
            isRecording = false;
            if (!statusMsgDiv.innerHTML.includes('Распознано') && !statusMsgDiv.innerHTML.includes('Текст очищен') && !statusMsgDiv.innerHTML.includes('Готов')) {
              setStatus('🎙️ Готов к новой записи');
            } else if (statusMsgDiv.innerHTML.includes('Ошибка') || statusMsgDiv.innerHTML.includes('не обнаружена')) {
              setStatus('🎙️ Нажмите микрофон, чтобы повторить');
            } else {
              // если статус уже "распознано" — оставляем, но можно дополнить
              if (!statusMsgDiv.innerHTML.includes('Распознано')) {
                setStatus('🎙️ Готов к записи');
              }
            }
          }
        };
        
        recogInstance.onstart = () => {
          isRecording = true;
          if (recordBtn) {
            recordBtn.classList.add('active');
            const micSpan = recordBtn.querySelector('.mic-icon');
            if (micSpan) micSpan.innerHTML = '⏺️';
          }
          setStatus('🔴 Запись... Говорите чётко');
        };
      }
      
      // Функция запуска записи
      function startRecording() {
        if (!SpeechRecognitionAPI) {
          setStatus('❌ Speech Recognition не поддерживается. Обновите браузер.', true);
          return;
        }
        
        // Создаём новый экземпляр каждый раз, чтобы избежать проблем с повторным использованием
        const newRecognition = initRecognition();
        if (!newRecognition) return;
        
        // Привязываем события
        bindRecognitionEvents(newRecognition);
        recognition = newRecognition;
        
        try {
          recognition.start();
        } catch (err) {
          console.error('Ошибка запуска распознавания:', err);
          setStatus('⚠️ Не удалось запустить запись. Попробуйте перезагрузить страницу.', true);
          if (recordBtn) recordBtn.classList.remove('active');
          const micSpan = recordBtn?.querySelector('.mic-icon');
          if (micSpan) micSpan.innerHTML = '🎤';
          isRecording = false;
        }
      }
      
      // Обработка клика по кнопке микрофона
      if (recordBtn) {
        recordBtn.addEventListener('click', (e) => {
          e.preventDefault();
          // Если уже идёт запись — останавливаем её
          if (isRecording) {
            if (recognition) {
              try {
                recognition.stop();
              } catch(stopperr) {
                // игнорим
              }
            }
            // Визуально сбросим состояние (обработчик onend тоже сбросит, но сделаем сразу для отзывчивости)
            recordBtn.classList.remove('active');
            const micSpan = recordBtn.querySelector('.mic-icon');
            if (micSpan) micSpan.innerHTML = '🎤';
            isRecording = false;
            setStatus('⏸️ Запись остановлена. Нажмите микрофон снова, чтобы продолжить.');
          } else {
            // Начинаем новую запись
            // Перед запуском проверим, не висит ли старый recognition, и уничтожим ссылку
            if (recognition) {
              try {
                recognition.abort();
              } catch(e) {}
              recognition = null;
            }
            startRecording();
          }
        });
      }
      
      // Дополнительно: если страница на мобильном, но API отсутствует - покажем предупреждение и деактивируем кнопку
      if (!SpeechRecognitionAPI) {
        setStatus('⚠️ Ваш браузер не поддерживает голосовой ввод. Используйте Chrome / Safari.', true);
        if (recordBtn) {
          recordBtn.disabled = true;
          recordBtn.style.opacity = '0.5';
          recordBtn.style.filter = 'grayscale(0.3)';
        }
      }
      
      // Небольшая подсказка для пользователя: при первом касании кнопки браузер запросит разрешение.
      // Эмуляция нормального поведения.
      
      // Очистка текста также сбрасывает currentTranscript
      // Добавим также обработчик на случай, если пользователь хочет начать с чистого листа
      
      // Полезная функция: если текущий сеанс завис, можно пересоздать
      window.addEventListener('pageshow', () => {
        if (isRecording && recognition) {
          // если страница стала видимой, а запись зависла - принудительно сбросим состояние кнопки
          if (recordBtn) recordBtn.classList.remove('active');
          const micSpan = recordBtn?.querySelector('.mic-icon');
          if (micSpan) micSpan.innerHTML = '🎤';
          isRecording = false;
          setStatus('🔄 Сессия обновлена. Готов к записи');
        }
      });
      
      // начальное состояние интерфейса
      updateTranscriptUI('');
      setStatus('🎙️ Нажмите на микрофон и разрешите доступ');
    }
    
    // Доп. корректировка отображения при изменении ориентации (чтобы заглушка оставалась корректной)
    window.addEventListener('resize', () => {
      // Если мы на десктопе, но разрешение экрана поменялось (например, повернули планшет? но десктопная заглушка не должна меняться)
      // Поскольку при загрузке уже определён тип устройства, переопределять не будем — это финальное решение.
      // Но для редких гибридных устройств можно перепроверить при сильном изменении ширины.
      // Но во избежание ложных переходов, лучше оставить как есть.
    });
  })();
</script>
